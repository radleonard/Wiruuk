<script>
function hitungNilai() {
    // Ambil nilai input
    let bara = parseFloat(document.getElementById('bara').value) || 0;
    let waktu = parseFloat(document.getElementById('waktu').value) || 0;
    let bowl = parseInt(document.getElementById('teori_bowl').value) || 0;
    let reb_benar = parseInt(document.getElementById('rebutan_benar').value) || 0;
    let reb_salah = parseInt(document.getElementById('rebutan_salah').value) || 0;
    let izin_salah = parseInt(document.getElementById('tanpa_izin').value) || 0;
    let manajemen = parseFloat(document.getElementById('manajemen').value) || 0;
    let sikap = parseFloat(document.getElementById('sikap').value) || 0;

    // Logika Ketepatan Waktu
    let skor_waktu = (waktu < 8) ? 10 : 9;

    // Logika Teori
    let skor_teori = (bowl * 4) + (reb_benar * 2) - (reb_salah * 1) - (izin_salah * 1);
    if (skor_teori < 0) skor_teori = 0; // Mengunci agar skor teori tidak minus

    // Kalkulasi Berbobot
    let n_bara = bara * 0.30;
    let n_waktu = skor_waktu * 0.25;
    let n_teori = skor_teori * 0.25;
    let n_manajemen = manajemen * 0.10;
    let n_sikap = sikap * 0.10;

    let nilai_akhir = n_bara + n_waktu + n_teori + n_manajemen + n_sikap;

    // Tampilkan Hasil
    document.getElementById('detail_skor').innerHTML = `
        • Skor Waktu: ${skor_waktu} poin (Durasi: ${waktu} mnt)<br>
        • Skor Teori Murni: ${skor_teori} poin<br>
        <small style="color:#7f8c8d;">*Perhitungan Teori: (Bowl:${bowl}x4) + (Reb Benar:${reb_benar}x2) - (Reb Salah:${reb_salah}) - (Tanpa Izin:${izin_salah})</small>
    `;
    document.getElementById('total_skor').innerHTML = `NILAI AKHIR: ${nilai_akhir.toFixed(2)}`;
    document.getElementById('hasil').style.display = 'block';
}
</script>
