<script>
// Fungsi mengatur tampilan jumlah form peserta (3 atau 4)
function aturJumlahPeserta() {
    let jumlah = document.getElementById('jumlah_peserta').value;
    if (jumlah == "3") {
        document.getElementById('card-4').style.display = 'none';
    } else {
        document.getElementById('card-4').style.display = 'flex';
    }
}

// Fungsi utama perhitungan skor per peserta
function hitungSkor(id) {
    // 1. Ambil semua Nilai dari Elemen
    let bara = parseFloat(document.getElementById(`bara-${id}`).value);
    let waktu = parseFloat(document.getElementById(`waktu-${id}`).value);
    let bowl = parseInt(document.getElementById(`bowl-${id}`).value) || 0;
    let reb_b = parseInt(document.getElementById(`rebutan-b-${id}`).value) || 0;
    let reb_s = parseInt(document.getElementById(`rebutan-s-${id}`).value) || 0;
    let izin = parseInt(document.getElementById(`izin-${id}`).value) || 0;
    let m_5r = parseFloat(document.getElementById(`5r-${id}`).value);
    let sikap = parseFloat(document.getElementById(`sikap-${id}`).value);

    // 2. Update Label Teks Bar Level / Slider secara Real-Time
    document.getElementById(`lbl-bara-${id}`).innerText = bara;
    document.getElementById(`lbl-waktu-${id}`).innerText = waktu.toFixed(1) + " Menit";
    document.getElementById(`lbl-5r-${id}`).innerText = m_5r;
    document.getElementById(`lbl-sikap-${id}`).innerText = sikap;

    // 3. Logika Poin Waktu (Dibawah 8 menit = 10, di atas/sama dengan 8 menit = 9)
    let poin_waktu = (waktu < 8) ? 10 : 9;

    // 4. Logika Poin Teori murni
    let poin_teori = (bowl * 4) + (reb_b * 2) - (reb_s * 1) - (izin * 1);
    if (poin_teori < 0) poin_teori = 0; // Kunci batas bawah 0 agar tidak minus

    // 5. Kalkulasi Menggunakan Bobot Persentase
    let n_bara = bara * 0.30;
    let n_waktu = poin_waktu * 0.25;
    let n_teori = poin_teori * 0.25;
    let n_5r = m_5r * 0.10;
    let n_sikap = sikap * 0.10;

    let nilai_akhir = n_bara + n_waktu + n_teori + n_5r + n_sikap;

    // 6. Cetak Hasil Akhir ke dalam Kotak Skor masing-masing peserta
    document.getElementById(`total-${id}`).innerText = nilai_akhir.toFixed(2);
    document.getElementById(`info-waktu-${id}`).innerText = `Poin Waktu: ${poin_waktu} | Poin Teori Murni: ${poin_teori}`;
}

// Inisialisasi hitungan awal saat aplikasi pertama kali dibuka
for (let i = 1; i <= 4; i++) {
    hitungSkor(i);
}
aturJumlahPeserta();
</script>
